#include <iostream>

using namespace std;

int main()
{
    int n, max, kmax, max1, i;
    cin>>n;
    int k = rand() % n;
    int *arr = new int[n];
    for(int i = 0; i < n; i++){
        arr[i] = i+1;
    }
    for (int i = 0; i < n-1; i++)
    {
        int j = rand() % (n - i) + i;
        int temp = arr[j]; 
        arr[j] = arr[i]; 
        arr[i] = temp; 
    }
    cout<<"Исходный массив:"<<endl;
    for(int i = 0; i < n; i++){
        cout<<arr[i]<<" ";
    }
    for (i = 0; i < n; i++)
    {
        if (arr[i] > max) {
            max = arr[i];
            kmax = i;
        }
    }
    for (i = 0; i < n; i++)
    {
        if (arr[i] > max1 && i != kmax) {
            max1 = arr[i];
        }
    }
    cout<<endl<<"2 максиума: "<<max<<", "<<max1;
}
